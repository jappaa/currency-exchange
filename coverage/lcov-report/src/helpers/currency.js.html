const currencyDivide = ( currencyMaster, bank, currency, acronym ) =>
  currency.sell > 1
    ? [
      ...bank,
      {
        ...currency,
        buy: number.mathRound( currency.buy / 100, 5 ),
        sell: number.mathRound( currency.sell / 100, 5 ),
        extendName: null,
        currencyDetails: currencyMaster[ acronym ],
        indexGroup: 0,
      },
    ]
    : [
      ...bank,
      {
        ...currency,
        extendName: null,
        currencyDetails: currencyMaster[ acronym ],
        indexGroup: 0,
      },
    ]
 
/** specific
