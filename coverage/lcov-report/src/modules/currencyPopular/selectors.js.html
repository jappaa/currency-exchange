import { createSelector } from 'reselect'
import uniq from 'lodash/uniq'
import { number } from '../../services'
import { currency } from '../../helpers'
 
const getBanks = state => state.entities.banks
const getBanksCurrencyPopular = state => state.feature.currencyPopular.banks
 
const getBanksName = createSelector( getBanks, banks =>
  banks.reduce( ( p, c ) => [ ...p, c.bankNameEng ], [] ) )
 
const getBanksCurrencies = createSelector( getBanks, banks =>
  banks.reduce(
    ( p, c ) => [
      ...p,
      ...c.data.reduce( ( p, c ) => [ ...p, c.currencyDetails.code ], [] ),
    ],
    [],
  ) )
 
const getBankscurrenciesUniq = createSelector( getBanksCurrencies, currencies =>
  uniq( currencies ) )
 
const getBankscurrenciesUniqToObj = createSelector(
  getBankscurrenciesUniq,
  currencies => currencies.map( e => ( { label: e, value: e } ) ),
)
 
// additional arguments
/**
   * @param  {[number]}    index   [number such as 1, 2, 3, 4]
   * @return {[array]}     banks   [return bank by index]
   */
const getBanksByIndex = ( state, index ) => getBanks( state )[ index ]
 
/**
   * @param  {[number]}    index   [number such as 1, 2, 3, 4]
   * @return {[array]}     banks   [return bank by index]
   */
const getBankCurrencyPopularByIndex = ( state, index ) =>
  getBanksCurrencyPopular( state )[ index ]
 
/**
   * @param  {[string]}    acronym   [USD]
   * @param  {[float]}     num       [money]
   * @return {[array]}     banks     [return all banks give your high rate bank selling by currency and calculate selling price]
   */
const getBanksTopRateValue = ( state, currenciesMaster, acronym, num ) =>
  getBanks( state )
    .reduce(
      ( p, c, k ) => {
        // TODO [code fix]: USD case
        const sort = !c[ `${ acronym }` ] && acronym !== 'USD'
          ? [ currency.defaultCurrency( currenciesMaster, c, acronym ) ]
          : c.data
              .filter(
                e =>
                  acronym.toUpperCase().indexOf( e.currencyDetails.code ) !==
                    -1 && e.sell !== 0,
              )
              .map( e => ( {
                ...e,
                calSell: number.isNumber( num / e.sell, 0 ),
                calBuy: number.isNumber( num / e.buy, 0 ),
              } ) )
              .sort( ( a, b ) => b.calSell - a.calSell )
 
        // TODO USD case and duplicate
        const addFirstLevel = !c[ `${ acronym }` ] && acronym !== 'USD'
          ? currency.defaultCurrency( currenciesMaster, c, acronym )
          : sort[ 0 ]
        return [ ...p, { ...c, data: sort, ...addFirstLevel } ]
      },
      [],
    )
    .sort( ( a, b ) => b.calSell - a.calSell )
 
/**
   * @param  {[array]}    popularCurrencies  ['USD', 'JPJ', 'GBP']
   * @param  {[object]}   currenciesMaster   [currency detail give the detail symbol, name, etc...]
   * @return {[array]}    banks              [return bank sorted by index]
   */
const getBanksSortByCurrency = ( state, currenciesMaster, popularCurrencies ) =>
  getBanks( state ).reduce(
    ( p, c ) => {
      const popular = popularCurrencies.map(
        ( e, k ) =>
          !c[ e ]
            ? currency.defaultCurrency( currenciesMaster, c, e, k )
            : { ...c[ e ], index: k },
      )
      const old = c.data.reduce(
        ( p, c, k ) =>
          popularCurrencies.every( e => c.acronym.indexOf( e ) === -1 )
            ? [ ...p, { ...c, index: popular.length + p.length } ]
            : p,
        [],
      )
 
      return [ ...p, { ...c, data: [ ...popular, ...old ] } ]
    },
    [],
  )
 
/**
   * @param  {[number]}   index                     [number such as 1, 2, 3, 4]
   * @param  {[array]}    popularcurrenciesLists    ['USD', 'JPJ', 'GBP']
   * @return {[array]}    banks                     [return banks slice by popular currencies]
   */
const getBankCurrencyPopularPropsDataByPopular = (
  state,
  index,
  popularcurrenciesLists,
) =>
  getBankCurrencyPopularByIndex( state, index ).data.slice(
    0,
    popularcurrenciesLists.length,
  )
